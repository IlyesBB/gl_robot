def signe(x:int or float):
    if x > 0.0:
        return 1
    elif x < 0.0:
        return -1
    else:
        return 0

def positive_angle(angle:float or int):
    if signe(angle) < 0:
        return angle + (2*pi)
    else:
        return angle
